Python SDK

from aurelo import Aurelo
cli = Aurelo(base_url="https://api.aurelo.local", token="...")

# Create dataset and upload manifest
ds = cli.datasets.create(name="retail_text_2025Q3", modality="text")
cli.datasets.upload_manifest(ds.id, "manifest.jsonl")

# Launch labeling with policy
cli.labeling.start(dataset_version_id=ds.current_version_id,
                   policy={"llm_assist":True,"confidence_threshold":0.9})

# Train and register
run = cli.train.start(dataset_version_id=ds.current_version_id,
                      template="llm_finetune",
                      params={"lr":2e-5,"bf16":True})
model_v = cli.registry.register(run.id, name="aurelo-llm")
cli.deploy.canary(model_v.id, traffic=0.2)
